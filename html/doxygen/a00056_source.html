77 ALuint Openal::load(const char * filename)
78 {
79  SF_INFO fileinfos;
80  SNDFILE *file = sf_open(filename, SFM_READ, &fileinfos);
81  if (!file) return 0;
82  ALsizei nbsamples = static_cast<ALsizei>(fileinfos.channels *fileinfos.frames);
83  ALsizei samplerate = static_cast<ALsizei>(fileinfos.samplerate);
84  std::vector<ALshort> samples(nbsamples);
85  if (sf_read_short(file, &samples[0], nbsamples) < nbsamples) return 0;
86  sf_close(file);
87 
88  ALenum format;
89  switch (fileinfos.channels) {
90  case 1 : format = AL_FORMAT_MONO16; break;
91  case 2 : format = AL_FORMAT_STEREO16; break;
92  default : return 0;
93  }
94 
95  ALuint buffer;
96  alGenBuffers(1, &buffer);
97  alBufferData(buffer, format, &samples[0], nbsamples * sizeof(ALushort), samplerate);
98  if (alGetError() != AL_NO_ERROR) return 0;
99  return buffer;
100 }
