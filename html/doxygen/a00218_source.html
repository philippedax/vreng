173 static bool string2Coord(const char* positions, float& x, float& y, float& z, float& az)
174 {
175  char *positions_copy = strdup(positions);
176  char *p = null, *brkt = null;
177 
178  x = y = z = az = 0;
179  p = strtok_r(positions_copy, ",", &brkt);
180  if (p) x = (float) atof(p); else goto ERROR;
181  p = strtok_r(NULL, ",", &brkt);
182  if (p) y = (float) atof(p); else goto ERROR;
183  p = strtok_r(NULL, ",", &brkt);
184  if (p) z = (float) atof(p); else goto ERROR;
185  p = strtok_r(NULL, ",", &brkt);
186  if (p) az = (float) atof(p); else goto ERROR;
187 
188  free(positions_copy);
189  return true;
190 ERROR:
191  free(positions_copy);
192  return false;
193 }

249 void Messages::postRequest(const std::string& mess, std::string& result)
250 {
251  int sizeMax = 256;
252  int sizerecu = mess.length();
253  float tmpx, tmpy, tmpz, tmpaz; //coordonnees trouvees
254  int afficher = 0;
255  int nbObj = 0;
256  char posbuf[sizerecu +1];
257  char namebuf[sizerecu +1];
258  char tmpbuf[sizerecu +1];
259  char **listeObjets = new char*[sizeMax];
260 
261  posbuf[0] = '\0';
262  namebuf[0] = '\0';
263  tmpbuf[0] = '\0';
264 
265  for (int i=0; i<sizerecu ; i++) {
266  if (mess[i] == '@') {
267  if (nbObj > (sizeMax-2)) {
268  cerr << "message has too many request" <<endl;
269  break;
270  }
271  if (afficher == 0) {
272  posbuf[0] = '\0';
273  namebuf[0] = '\0';
274  }
275  else if (afficher == 1) {
276  listeObjets[nbObj] = strdup(namebuf);
277  nbObj++;
278  }
279  else if (afficher == 2) {
280  //on enregistre l'image ds un fichier.
281  listeObjets[nbObj] = strdup("request.jpg");
282  listeObjets[nbObj+1] = strdup(posbuf);
283 
284  //on convertit le string en coords.
285  string2Coord(posbuf, tmpx, tmpy, tmpz, tmpaz);
286 
287  //recupere la position et fait un snapshot de la position demande
288  ::g.render.calculateFov(tmpx,tmpy,tmpz,tmpaz, listeObjets[nbObj]);
289  nbObj += 2;
290  }
291  afficher = (afficher+1) % 3;
292  continue;
293  }
294  else { //not @
295  if (afficher == 0) {
296  //sprintf(msgAffiche, "%s%c", msgAffiche, mess[i]);
297  result += mess[i];
298  }
299  else if (afficher == 1) {
300  //sprintf(msgAffiche, "%s%c", msgAffiche, mess[i]);
301  result += mess[i];
302  strcpy(tmpbuf, namebuf);
303  sprintf(namebuf, "%s%c", tmpbuf, mess[i]);
304  }
305  else if (afficher == 2) {
306  strcpy(tmpbuf, posbuf);
307  sprintf(posbuf, "%s%c", tmpbuf, mess[i]);
308  }
309  }
310  }
311 
312  if (nbObj > 0) convertTextToLink(result, listeObjets, nbObj);
313  //TODO ev : traiter le message non augmente
314 
315  if (listeObjets) delete[] listeObjets;
316  listeObjets = NULL;
317 }

322 void Messages::writeMessage(const char* mode, const char* from, const char* msg)
323 {
324  if (!msg) return;
325 
326  uptr<UStr> text = null;
327  uptr<UStr> prefix = null;
328  uptr<UColor> prefix_color = null;
329 
330  if (from) {

332  postRequest(msg, result);
333  text = new UStr(result);
334 
335  prefix = new UStr("from ");
336  prefix->append(from);
337  prefix->append("> ");
338  prefix_color = &::g.theme.chatColor;
339  }
340 
341  else if (mode) {
342  text = new UStr(msg);
343 
344  if (! strcmp(mode, "notice")) {
345  prefix = new UStr(mode);
346  prefix->append("> ");
347  prefix_color = ::g.theme.noticeColor;
348  }
349  else if (! strcmp(mode, "progress")) {
350  prefix = new UStr("");
351  prefix_color = ::g.theme.noticeColor;
352  messages.add(uhbox(ugroup(UFont::bold + prefix_color) + text));
353  }
354  else if (! strcmp(mode, "request")) {
355  prefix = new UStr(mode);
356  prefix->append("> ");
357  prefix_color = ::g.theme.requestColor;
358  }
359  else {
360  prefix = new UStr(mode);
361  prefix->append("> ");
362  prefix_color = ::g.theme.warningColor;
363  }
364  }
365  else {
366  text = new UStr(msg);
367  prefix = new UStr("> ");
368  prefix_color = ::g.theme.messColor;
369  }
370 
371  messages.add(uhbox(ugroup(UFont::bold + prefix_color + prefix) + text));
372 
373  return;
