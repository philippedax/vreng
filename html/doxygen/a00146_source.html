163 /* Sets local world name */
164 void World::setName(const char *urlOrName)
165 {
166  // Find the begining of the last path component
167  const char* begin = strrchr(urlOrName, '/');
168  begin = (begin != NULL) ? begin+1 : urlOrName;
169 
170  // Find end of name by suppressing extention
171  const char* end = strrchr(begin, '.');
172  if (end == NULL) end = begin + strlen(begin);
173 
174  // Copy the name
175  if (name) delete name;
176  name = new char[end-begin + 1];
177  memcpy(name, begin, end-begin);
178  name[end-begin] = '\0';
179 
180  trace(DBG_WO, "setName: %s", name);
181 }

368 void World::compute(time_t sec, time_t usec)
369 {
370  uint16_t dimx, dimy, dimz;
371 
372  trace(DBG_WO, "compute:");
373  switch (getState()) {
374 
375  case LOADING:
376  //error("compute: no end encountered");
377  return;
378 
379  case LOADED:
380  trace(DBG_WO, "compute loaded:");
381  if (localuser) {
382  localuser->move.perm_sec = sec;
383  localuser->move.perm_usec = usec;
384  }
385  //notice("names: %d", namecnt);
386 
387  //
388  // computes world's bb
389  //
390  for (list<WObject*>::iterator it = stillList.begin(); it != stillList.end(); ++it) {
391  if (! (*it)->isValid()) continue;
392  if (! (*it)->bbBehavior() || (*it)->isBehavior(COLLIDE_NEVER)) continue;
393  for (int i=0; i<3 ; i++) {
394  bbmin.v[i] = MIN(bbmin.v[i], (*it)->pos.bbcenter.v[i] - (*it)->pos.bbsize.v[i]);
395  bbmax.v[i] = MAX(bbmax.v[i], (*it)->pos.bbcenter.v[i] + (*it)->pos.bbsize.v[i]);
396  }
397  }
398  for (list<WObject*>::iterator it = mobileList.begin(); it != mobileList.end(); ++it) {
399  if (! (*it)->isValid()) continue;
400  if (! (*it)->bbBehavior() || (*it)->isBehavior(COLLIDE_NEVER) || (*it)->type == 1) continue;
401  for (int i=0; i<3 ; i++) {
402  bbmin.v[i] = MIN(bbmin.v[i], (*it)->pos.bbcenter.v[i] - (*it)->pos.bbsize.v[i]);
403  bbmax.v[i] = MAX(bbmax.v[i], (*it)->pos.bbcenter.v[i] + (*it)->pos.bbsize.v[i]);
404  }
405  if (bbmax.v[0] > 1000 || bbmax.v[1] >1000 || bbmax.v[2] > 1000)
406  error("mobil: %d %s bbmin=%.1f,%.1f,%.1f bbmax=%.1f,%.1f,%.1f", (*it)->type, (*it)->getInstance(), bbmin.v[0], bbmin.v[1], bbmin.v[2], bbmax.v[0], bbmax.v[1], bbmax.v[2]);
407  }
408  for (int i=0; i<3 ; i++) {
409  bbcenter.v[i] = (bbmax.v[i] + bbmin.v[i]);
410  bbsize.v[i] = (bbmax.v[i] - bbmin.v[i]);
411  }
412  notice("size=%.1f,%.1f,%.1f center=%.1f,%.1f,%.1f", bbsize.v[0], bbsize.v[1], bbsize.v[2], bbcenter.v[0], bbcenter.v[1], bbcenter.v[2]);
413 
414  OList::clearIspointedFlag(mobileList);
415 
416  // compute Grid dimensions
417  dimx = (int) (bbsize.v[0] / DISTX);
418  dimy = (int) (bbsize.v[1] / DISTY);
419  dimz = (int) (bbsize.v[2] / DISTZ);
420  dimx = (dimx % 2) ? dimx + 1 : dimx;
421  dimy = (dimy % 2) ? dimy + 1 : dimy;
422  dimz = (dimz % 2) ? dimz + 1 : dimz;
423  dimx = MIN(64, dimx);
424  dimy = MIN(64, dimy);
425  dimz = MIN(16, dimz);
426  //notice("dim: %d,%d,%d", dimx, dimy, dimz);
427  //dimgrid[] = dimx;
428  //dimgrid[1] = dimy;
429  //dimgrid[2] = dimz;
430  // free the current grid, reinitialize another one with new dimensions
431  //World::current->freeGrid();
432  //World::current->initGrid(dimgrid, slice);
433 
434  Grid::grid()->init(dimx, dimy, dimz);
435  Axis::axis()->init();
436  setState(SIMULATION);
437  return;
438 
439  case SIMULATION:
440  trace(DBG_WO, "compute simulation:");
441  //
442  // user motions
443  //
444  if (! localuser) return;
445  if (! localuser->isValid()) return;
446  if (::g.pref.dbgtrace) error("user mov: %s-%s", localuser->typeName(), localuser->named());
447  localuser->userMovement(sec, usec);
448 
449  //
450  // removes objects scheduled to be deleted
451  //
452  deleteObjects();
453 
454  //
455  // objects with imposed and permanent motions
456  //
457  {
458  int i=0;
459  for (list<WObject*>::iterator it = mobileList.begin(); it != mobileList.end(); ++it, i++) {
460  if (! (*it)->isValid()) {
461  error("bad type=0: %d", i);
462  mobileList.remove(*it);
463  continue;
464  }
465  if ((*it)->type > OBJECTSNUMBER) {
466  error("bad type out of range: %d t=%d", i, (*it)->type);
467  mobileList.remove(*it);
468  continue;
469  }
470  if ((*it)->mode > 2) {
471  error("bad mode: %d m=%d", i, (*it)->mode);
472  mobileList.remove(*it);
473  continue;
474  }
475  if (! (*it)->num) {
476  error("num null: %d", i);
477  mobileList.remove(*it);
478  continue;
479  }
480  if (::g.pref.dbgtrace) error("obj: %s-%s", (*it)->typeName(), (*it)->getInstance());
481  (*it)->imposedMovement(sec, usec);
482  (*it)->permanentMovement(sec, usec);
483  }
484  }
485  default: return;
486  }
487 }

489 /* Switch channels */

491 bool World::call(World *wpred)
492 {
493  trace(DBG_WO, "call: leave chan=%s", wpred->chan);
494  if (wpred->islinked) {
495  enter(url, NULL, OLD);
496  setChan(wpred->chan);
497  }
498  else {
499  trace(DBG_IPMC, "call: leave chan=%s", wpred->chan);
500  if (Channel::current()) delete Channel::current(); // leave current channel
501 
502  enter(url, NULL, OLD);
503 
504  char groupstr[GROUP_LEN];
505  Channel::getGroup(chan, groupstr);
506  group = inet_addr(groupstr);
507 
508  trace(DBG_IPMC, "call: join chan=%s", chan);
509  if (Channel::join(chan) == 0) { // join previous channel
510  trace(DBG_IPMC, "call: can't join chan=%s", chan);
511  return false;
512  }
513  setChan(chan);
514 
515  ::g.gui.updateWorld(this, NEW); // nofify the gui
516  }
517  return true;
518 }

522 World * World::goBack()
523 {
524  World *world = worldList;
525  World *worldback = worldList->next;
526 
527  if (! worldback) return NULL;
528  trace(DBG_WO, "goBack: ");
529 
530  world->quit(); // quit current world first
531 
532  World *wp;
533  int loop = 0;
534  for (wp = worldback; wp->next && (loop <100); wp = wp->next, loop++)
535  if (wp == wp->next) break;
536 
537  wp->next = world;
538  world->next = NULL;
539  worldback->prev = NULL;
540  world->prev = worldback;
541  worldList = worldback;
542  //debug dumpworldlist("back");
543

574 /* Exchange Worlds in the list */
575 // static
576 World * World::swap(World *w)
577 {
578  if (worldList == w) return worldList;
579 
580  trace(DBG_WO, "swap: ");
581  if (w->prev)
582  w->prev->next = worldList; // 1
583  if (w->next)
584  w->next->prev = worldList; // 2
585  if (worldList->next)
586  worldList->next->prev = w; // 3
587  worldList->prev = w->prev; // 4
588  World *tmp = worldList->next;
589  worldList->next = w->next; // 5
590  w->next = tmp; // 6
591  w->prev = NULL; // 7
592  worldList = w; // 8
593  return worldList;
594 }

685 /* Check and load my proper icons - static */
686 void World::checkIcons()
687 {
688  chdir(::g.env.icons());
689  DIR *dirw = opendir(".");
690  if (dirw) {
691  // find if current world is there
692  for (struct dirent *dw = readdir(dirw); dw; dw = readdir(dirw)) {
693  struct stat bufstat;
694  if (stat(dw->d_name, &bufstat) == 0 &&
695  S_ISDIR(bufstat.st_mode) &&
696  ! strcmp(dw->d_name, getName())) {
697  chdir(dw->d_name);
698  DIR *diri = opendir(".");
699  if (diri) {
700  // find icons in this world
701  for (struct dirent *di = readdir(diri); di; di = readdir(diri)) {
702  if (stat(di->d_name, &bufstat) == 0 &&
703  S_ISREG(bufstat.st_mode)) {
704  // open the icon and read it
705  FILE *fp;
706  if ((fp = File::openFile(di->d_name, "r")) == NULL) {
707  error("can't open %s/%s/%s", ::g.env.icons(), getName(), di->d_name);
708  continue;
709  }
710  char vref[BUFSIZ], infos[BUFSIZ *2], urlvre[URL_LEN];
711  fgets(vref, sizeof(vref), fp);
712  File::closeFile(fp);
713  Cache::file2url(di->d_name, urlvre);
714  // create the icon
715  sprintf(infos, "<url=\"%s\">&<vref=%s>", urlvre, vref);
716  trace(DBG_WO, "load-icon: %s", infos);
717  if (isAction(ICON_TYPE, Icon::CREATE))
718  doAction(ICON_TYPE, Icon::CREATE, localUser(), infos, 0, 0);
719  }
720  }
721  closedir(diri);
722  }
723  break;
724  }
725  }

792 
793  FILE *vrefp;

795  struct stat vrestat;

797

800  error("can't create file %s", vrefile);

802  // download the vre file from the httpd server
803 httpread:
804  while ((vrelen = http->httpRead(vrebuf, sizeof(vrebuf))) > 0) {

808  break;
809  }
810 
811 #if 0 //HAVE_LIBXML2
812  Xml::dtdValidation(vrefile); // check the DTD
813 #endif //HAVE_LIBXML2
814  }
815  else { // file exists in the cache
816  if ((vrefp = File::openFile(vrefile, "r")) == NULL) goto httpread; // download it
817  while ((vrelen = fread(vrebuf, 1, sizeof(vrebuf), vrefp)) > 0) {
818  if (parser->parseVreLines(vrebuf, vrelen) <= 0) // parsing error
819  break;
820  }
821  }
822  parser->numline = 0;
823  if (vrefp) File::closeFile(vrefp);
824

826  return;
