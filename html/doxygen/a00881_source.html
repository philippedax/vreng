293 /* Builds a netobject name from the string "scene_id/obj_id" */
294 void NetObject::buildNoidFromString(const char *s, bool netbehave)
295 {
296  uint16_t scene_id, obj_id;
297  int c = sscanf(s, "%hu/%hu", &scene_id, &obj_id);
298 
299  if (c != 2 || scene_id == 0) {
300  error("buildNoidFromString: invalid name %s c=%d scene=%d", s, c, scene_id);
301  return;
302  }
303 
304  permanent = netbehave; // should be true
305  initProperties(false); // we are not responsible
306 
307  noid.src_id = htonl(1);
308  noid.port_id = htons(scene_id);
309  noid.obj_id = htons(obj_id);
310 
311  if (getNetObject()) {
312  warning("buildNoidFromString: %s already seen %d/%d", pobject->getInstance(), scene_id, obj_id);
313  return;
314  }
315  addToList();
316 }
