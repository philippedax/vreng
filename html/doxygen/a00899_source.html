61 /* Fills the payload */
62 int Payload::putPayload(const char *format, ...)
63 {
64  va_list ap;
65 
66  if (! isValidPayload()) { error("putPayload: invalid Payload %s", this); return -1; }
67  if (format == NULL) { error("putPayload: NULL format"); return -1; }
68 
69  va_start(ap, format);
70  len = idx; // "rewrite" mode rather than "append"
71  trace(DBG_NET, "putPayload: idx=%d", idx);
72 
73  /* Parse format */
74  while (*format) {
75  switch (*format) {
76  case 'c': // char
77  {
78  uint8_t c = va_arg(ap, int);
79 
80  data[idx++] = 'c';
81  data[idx++] = c;
82  break;
83  }
84  case 'h': // short
85  {
86  int16_t h = htons(va_arg(ap, int));
87 
88  data[idx++] = 'h';
89  memcpy(data + idx, &h, sizeof(int16_t));
90  idx += sizeof(int16_t);
91  break;
92  }
93  case 'd': // int32_t
94  {
95  int32_t d = htonl(va_arg(ap, int));
96 
97  data[idx++] = 'd';
98  memcpy(data + idx, &d, sizeof(int32_t));
99  idx += sizeof(int32_t);
100  break;
101  }
102  case 'f': // float
103  {
104  float f = va_arg(ap, double);
105  int32_t *p = (int32_t*) &f;
106  int32_t j = htonl(*p);
107 
108  data[idx++] = 'f';
109  memcpy(data + idx, &j, sizeof(int32_t));
110  idx += sizeof(int32_t);
111  break;
112  }
113  case 's': // string
114  {
115  uint8_t *s = va_arg(ap, uint8_t *);
116  uint16_t l;
117  l = (*s) ? strlen((char *) s) : 0;
118 
119  data[idx++] = 's';
120  data[idx++] = l;
121  if (l)
122  memcpy(data + idx, s, l);
123  idx += l;
124  break;
125  }
126  case 'n': // Noid
127  {
128  Noid n = va_arg(ap, Noid);
129 
130  data[idx++] = 'n';
131 
132  /* everything is already in network format */
133  memcpy(data + idx, &n.src_id, sizeof(n.src_id));
134  idx += sizeof(n.src_id);
135  memcpy(data + idx, &n.port_id, sizeof(n.port_id));
136  idx += sizeof(n.port_id);
137  memcpy(data + idx, &n.obj_id, sizeof(n.obj_id));
138  idx += sizeof(n.obj_id);
139  break;
140  }
141  case 't': // timeval
142  {
143  struct timeval t = va_arg(ap, struct timeval);
144  time_t sec = htonl(t.tv_sec);
145  time_t usec = htonl(t.tv_usec);
146 
147  data[idx++] = 't';
148  memcpy(data + idx, &sec, sizeof(int32_t));
149  idx += sizeof(int32_t);
150  memcpy(data + idx, &usec, sizeof(int32_t));
151  idx += sizeof(int32_t);
152  break;
153  }
154 
155  default: // unknown type in format
156  error("putPayload: invalid format '%c' in %s", *format, format);
157  va_end(ap);
158  return -1;
159  }
160 
161  format++;
162  if (idx > len)
163  len = idx;
164 
165  /* check the length */
166  if (len >= PAYLOAD_LEN) {
167  error("putPayload: Payload too long (%d > %d)", len, PAYLOAD_LEN);
168  dumpPayload(stderr);
169  len = idx = 0; // just in case
170  va_end(ap);
171  return -1;
172  }
173  }
174  va_end(ap);
175  return 0;
176 }

316

319  error("getPayload: past end of Payload: idx=%d len=%d", idx, len);
320  dumpPayload(stderr);
321  idx = len = 0;
322  va_end(ap);
323  return -1;
324  }
325  format++;
326  }
327  va_end(ap);
328  return 0;

341 int Payload::tellStrInPayload(const char *str)
342 {
343  uint16_t save_idx = idx;
344 
345  for (idx = 0; idx < len; ) {
346  uint8_t format = data[idx];
347 
348  /* checks format */
349  if (strchr("chdfsnt", format) == NULL) {
350  error("tellStrInPayload: invalid format [%c]", format);
351  idx = save_idx;
352  return -1;
353  }
354  idx++; // skip format
355 
356  switch (format) {
357  case 'c': // char
358  idx++;
359  break;
360  case 'h': // int16_t
361  idx += sizeof(int16_t);
362  break;
363  case 'd': // int32_t
364  case 'f': // float
365  idx += sizeof(int32_t);
366  break;
367  case 's': // string
368  if (!strncmp((char *) (data + idx + 1), str, strlen(str))) {
369  idx--; // string matches
370  return idx;
371  }
372  idx += data[idx];
373  idx++;
374  break;
375  case 'n': // Noid
376  idx += sizeof(uint32_t);
377  idx += sizeof(uint16_t);
378  idx += sizeof(uint16_t);
379  break;
380  case 't': // timeval
381  idx += sizeof(int32_t);
382  idx += sizeof(int32_t);
383  break;
384  default:
385  if (format != 0) {
386  error("tellStrInPayload: format unimplemented [%c 0x%02x]", format, format);
387  dumpPayload(stderr);
388  }
389  idx = save_idx;
390  return -1;
391  }
392  }
393  error("tellStrInPayload: past end of Payload: idx=%d len=%d", idx, len);
394  dumpPayload(stderr);
395  idx = save_idx;
396  return -1;
397 }

434 int Payload::sendPayload(const struct sockaddr_in *to)
435 {
436  uint8_t pkt[PKTSIZE];
437  uint8_t *hdrpl;
438  int pkt_len, sd;
439  rtp_hdr_t *rtp_hdr = (rtp_hdr_t *) pkt;
440 
441  /*
442  * initial checks
443  */
444  if (!isValidPayload()) { error("sendPayload: invalid"); return -1; }
445  if (!len) { trace(DBG_NET, "sendPayload: empty"); return -1; }
446  if (!to) { trace(DBG_NET, "sendPayload: to NULL"); return -1; }
447 
448  /*
449  * build the RTP Header
450  */
451  Channel *pchan;
452  if ((pchan = Channel::getbysa(to)) == NULL) {
453  pchan = Channel::current(); // hack !!!
454  error("sendPayload: pchan NULL, to=%p", to);
455  return -1; //FIXME: channel NULL
456  }
457  if (pchan->session) pchan->session->buildRtpHeader(rtp_hdr, pchan->ssrc);
458  else return -1; //FIXME: session NULL
459 
460  /*
461  * build the VREP Header
462  */
463  hdrpl = pkt + RTP_HDR_SIZE;
464 
465  hdrpl[VREP_HDR_VERSION] = vrep;
466  switch (vrep) {
467  case VREP_VERSION_V3:
468  pkt_len = RTP_HDR_SIZE + VREP_HDR_SIZE_V3 + len; // bytes
469  memcpy(hdrpl+VREP_HDR_SIZE_V3, data, len);
470  //TODO: if (gzipped) do compression
471  break;
472  case VREP_VERSION_V2:
473  hdrpl[VREP_HDR_FLAGS] = 1; // payload header length in words
474  hdrpl[VREP_HDR_IP] = AF_INET;
475  hdrpl[VREP_HDR_PLLEN] = (len + 3 + VREP_HDR_SIZE_V2) >> 2; // words
476  pkt_len = RTP_HDR_SIZE + (hdrpl[VREP_HDR_PLLEN] <<2); // bytes
477  memcpy(hdrpl+VREP_HDR_SIZE_V2, data, len);
478  hdrpl[VREP_HDR_SIZE_V2 + len] = 0; // end of payload chunk
479  break;
480  default:
481  error("bad vrep_version=%d", vrep); return -1;
482  }
483 
484  trace(DBG_NET, "S: %02x%02x%02x%02x/%02x",
485  hdrpl[0], hdrpl[1], hdrpl[2], hdrpl[3], hdrpl[4]);
486 
487  /*
488  * find the file descriptor
489  */
490  if ((sd = Channel::getFdSendRTP(to)) <= 0)
491  sd = pchan->sd[SD_W_RTP]; // hack !!!
492 
493  /*
494  * send the packet
495  */
496  Rtp::sendPacket(sd, pkt, pkt_len, to);
497  statSendRTP(pkt_len);
498 
499  /* Update SR, TODO: compute RCTP interval */
500  pchan->session->sr.psent++;
501  pchan->session->sr.osent += pkt_len;
502 
503  uint32_t sent = getSentPackets();
504  if (sent && ((sent % 100) == 0)) {
505  /*
506  * send RTCP compound (SR + SDES)
507  */
508  pchan->session->refreshMySdes();
509  pchan->session->sendSRSDES(to);
510  }
511 
512  return pkt_len;
513 }

518 int Payload::recvPayload(int fd, struct sockaddr_in *from)
519 {
520  if (! from) { error("recvPayload: from NULL"); return -1; }
521 
522  uint8_t pkt[PKTSIZE];
523  rtp_hdr_t *rtp_hdr = (rtp_hdr_t *) pkt;
524  rtcp_common_t *rtcp_hdr = (rtcp_common_t *) pkt;
525  int pkt_len;
526  socklen_t l = sizeof(struct sockaddr_in);
527 
528  memset(from, 0, sizeof(struct sockaddr_in));
529  if ((pkt_len = recvfrom(fd, pkt, sizeof(pkt), 0, (struct sockaddr *)from, &l)) <0) {
530 #if IPMC_ENABLED
531  error("recvfrom: %s on %d", strerror(errno), fd);
532 #endif
533  return pkt_len; // here pkt_len < 0 -> error
534  }
535  //trace(DBG_FORCE, "recvfrom: %lx (%x)", ntohl(from->sin_addr.s_addr), pkt_len);
536 
537  /*
538  * test if the packet was sent by myself
539  * this is probably broken anyway, loopback or not
540  */
541  if (NetObject::getMyHostId() == ntohl(from->sin_addr.s_addr)) {
542 #if NEEDLOOPBACK
543  // If two apps are running on the same machine,
544  // you need to sort out the packets on something else than just the host ID
545  if (ntohl(rtp_hdr->ssrc) == NetObject::getMySsrcId())
546 #endif
547  return 0; // Loopback from same app : ignore it
548  }
549 
550  statReceivePacket(pkt_len); //FIXME! is not a the good place, do statDroppedPacket
551 
552  /*
553  * test if it is a valid RTP header
554  */
555  uint8_t rtp_hdr_size;
556  if (rtp_hdr->version == RTP_VERSION &&
557  (rtp_hdr->pt == RTP_VREP_TYPE || rtp_hdr->pt == RTP_VREP_TYPE_V1)) {
558  //
559  // it's a RTP packet
560  //
561  rtp_hdr_size = RTP_HDR_SIZE;
562  uint16_t seq = ntohs(rtp_hdr->seq);
563  Source *pso;
564 
565  if ((pso = Source::getSource(ntohl(rtp_hdr->ssrc))) != NULL) {
566  sourceInfos *s;
567 
568  pso->rr.lost += seq - pso->rr.last_seq - 1;
569  pso->rr.last_seq = seq;
570  s = &(pso->s);
571  Rtp::updateSeq(s, seq);
572  pso->extended_max = s->cycles + s->max_seq;
573  pso->expected = pso->extended_max - s->base_seq + 1;
574  pso->lost = pso->expected - s->received; // number of packets lost
575  }
576  }
577  else if (rtcp_hdr->version == RTP_VERSION) {
578  //
579  // it's a RTCP packet
580  //
581  switch (rtcp_hdr->pt) {
582  case RTCP_SR:
583  case RTCP_SDES:
584  case RTCP_BYE:
585  case RTCP_APP:
586  return Rtp::recvRTCPPacket(from, pkt, pkt_len);
587  case RTCP_RR: // why send a RR ? we only send SR
588  return 0; // like action done
589  default:
590  return -3; // unknown RTCP packet type
591  }
592  }
593  else {
594  // it's an UDP packet, old version of VREng <= 1.5.8
595  rtp_hdr_size = 0;
596  }
597 
598  uint8_t *hdrpl = pkt + RTP_HDR_SIZE;
599  //trace(DBG_14, "R: %02x%02x%02x%02x/%02x",hdrpl[0],hdrpl[1],hdrpl[2],hdrpl[3],hdrpl[4]);
600 
601  /*
602  * compatibility with older VREP Protocol
603  */
604  uint8_t vrep_version = hdrpl[VREP_HDR_VERSION]; // vrep version received
605  int32_t vrep_len; // vrep header size + payload size
606 
607  uint8_t vrep_hdr_size; // vrep header size
608 
609  switch (vrep_version) {
610  case VREP_VERSION_V3:
611  vrep_hdr_size = VREP_HDR_SIZE_V3;
612  vrep_len = pkt_len - rtp_hdr_size - vrep_hdr_size;
613  gzipped = hdrpl[VREP_HDR_VERSION] & VREP_GZIP_MASK;
614  //TODO: ungzip data and update len
615  break;
616  case VREP_VERSION_V2:
617  /* old version <= 5.1.5 VREP2 */
618  vrep_hdr_size = VREP_HDR_SIZE_V2;
619  if (hdrpl[VREP_HDR_FLAGS] == 1)
620  vrep_len = hdrpl[VREP_HDR_PLLEN] << 2; // length in words
621  else
622  vrep_len = hdrpl[VREP_HDR_PLLEN]; // length in bytes
623  break;
624  default:
625  /* old version <= 1.5.8 VREP1 */
626  vrep_hdr_size = VREP_HDR_SIZE_V1;
627  vrep_len = hdrpl[VREP_HDR_PLLEN_V1]; // length in bytes <= 1.5.8 VREP1
628  memcpy(&from->sin_addr.s_addr, hdrpl+VREP_HDR_SRCID_V1, 4);
629  memcpy(&from->sin_port, hdrpl+VREP_HDR_PORTID_V1, 2);
630  }
631 
632  /*
633  * check if valid payload header
634  */
635  if (pkt_len != rtp_hdr_size + vrep_hdr_size + vrep_len || vrep_len < vrep_hdr_size) {
636  trace(DBG_NET, "R: %02x%02x%02x%02x %02x%02x%02x%02x %02x%02x%02x%02x",
637  pkt[0], pkt[1], pkt[2], pkt[3],
638  pkt[4], pkt[5], pkt[6], pkt[7],
639  pkt[8], pkt[9], pkt[10], pkt[11]);
640  trace(DBG_NET, "R: %02x%02x%02x%02x/%02x",
641  hdrpl[0], hdrpl[1], hdrpl[2], hdrpl[3], hdrpl[4]);
642  trace(DBG_NET, "wrong packet size, pkt_len=%d vrep_len=%d",
643  pkt_len, vrep_len);
644  return -2; // unknown packet type
645  }
646 
647  /*
648  * fill the Payload fields
649  */

658 /* Incoming Delta */
659 void Payload::incomingDelta(const struct sockaddr_in *from)
660 {
661  Noid noid; // name received
662  uint8_t prop_id; // property number received
663  int16_t vers_id; // version received
664 
665  if (getPayload("nch", &noid, &prop_id, &vers_id) < 0) return;
666 
667  NetObject *pn;
668  if ((pn = noid.getNetObject()) == NULL) {
669  /* delta on an unknown object */
670  trace(DBG_NET, "incomingDelta sendQuery on: %s, from=%s, p=%d, v=%d",
671  noid.getNetNameById(), inet4_ntop(&from->sin_addr), prop_id, vers_id);
672  // send a Query to the sender in unicast
673  noid.sendQueryNoid(from);
674  return;
675  }
676 
677  /* verify prop_id */
678  uint8_t nprop = pn->getPropertiesNumber();
679  if (prop_id >= nprop) {
680  error("incomingDelta: invalid property prop_id=%d"
681  "(type=%d, nprop=%d)", prop_id, pn->type, nprop);
682  return;
683  }
684  NetProperty *pprop = pn->netprop + prop_id;
685 
686  /*
687  * depends on version
688  */
689  // in complement to 2: d gives the distance, same throught the boundary
690  int16_t d = pprop->version - vers_id; // versions difference
691  if (abs(d) > 5000) { // very far
692  warning("incomingDelta: very different versions: mine is %d, received %d",
693  pprop->version, vers_id);
694  }
695  if (d > 0) return; // mine is more recent
696 
697  pprop->resetDates();
698  if (d < 0) { // mine is less recent, its own is more recent
699  pn->getProperty(prop_id, this);
700  pprop->setResponsible(false); // leave responsibility to the other one
701  pprop->version = vers_id;
702  return;
703  }
704  /* same versions, 2 responsibles: conflict */
705  else if (pprop->responsible) {
706  // resolved by getting a new random version
707  // publishes new version to sender in unicast
708  pn->declareObjDelta(prop_id);
709  // warning("Conflict resol: obj=%s, prop=%d, changing vers_num %d->%d",
710  // pn->getNetNameById(), prop_id, vers_id, pn->prop[prop_id].version);
711  }
712  // else, it's just a "recall" (then nothing to do)
713 }
