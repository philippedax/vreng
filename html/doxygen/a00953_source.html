139 /*
140  * Draw a box (taken from Mesa)
141  */
142 void Draw::box(GLfloat width, GLfloat depth, GLfloat height, const int textures[], GLfloat rtx[6][2], uint8_t style)
143 {
144  GLfloat x1 = -width/2;
145  GLfloat x2 = width/2;
146  GLfloat y1 = -depth/2;
147  GLfloat y2 = depth/2;
148  GLfloat z1 = -height/2;
149  GLfloat z2 = height/2;
150 
151  static const GLfloat normals[6][3] = {
152  { -1, 0, 0 },
153  { 0, 1, 0 },
154  { 1, 0, 0 },
155  { 0, -1, 0 },
156  { 0, 0, 1 },
157  { 0, 0, -1 }
158  };
159  static const int faces[6][4] = {
160  { 0, 1, 2, 3 },
161  { 3, 2, 6, 7 },
162  { 7, 6, 5, 4 },
163  { 4, 5, 1, 0 },
164  { 5, 6, 2, 1 },
165  { 7, 4, 0, 3 }
166  };
167  // Repeat Texture: GLfloat rtx[6][2] is an array of GLfloat containing
168  // nb. repeat for each texture's s-axis and t-axis
169  GLfloat texcoords[6][4][2] = {
170  { {0 ,rtx[0][1]},
171  {0 ,0} ,
172  {rtx[0][0],0} ,
173  {rtx[0][0],rtx[0][1]} },
174  { {rtx[1][0],rtx[1][1]},
175  {rtx[1][0],0} ,
176  {0 ,0} ,
177  {0 ,rtx[1][1]} },
178  { {0 ,rtx[2][1]},
179  {0 ,0} ,
180  {rtx[2][0],0} ,
181  {rtx[2][0],rtx[2][1]} },
182  { {rtx[3][0],rtx[3][1]},
183  {rtx[3][0],0} ,
184  {0 ,0} ,
185  {0 ,rtx[3][1]} },
186  { {0 ,0} ,
187  {0 ,rtx[4][1]},
188  {rtx[4][0],rtx[4][1]},
189  {rtx[4][0],0} },
190  { {0 ,0} ,
191  {0 ,rtx[5][1]},
192  {rtx[5][0],rtx[5][1]},
193  {rtx[5][0],0} }
194  };
195 
196  GLfloat v[8][3], tmp;
197  if (x1 > x2) { tmp = x1; x1 = x2; x2 = tmp; }
198  if (y1 > y2) { tmp = y1; y1 = y2; y2 = tmp; }
199  if (z1 > z2) { tmp = z1; z1 = z2; z2 = tmp; }
200 
201  v[0][0] = v[1][0] = v[2][0] = v[3][0] = x1;
202  v[4][0] = v[5][0] = v[6][0] = v[7][0] = x2;
203  v[0][1] = v[1][1] = v[4][1] = v[5][1] = y1;
204  v[2][1] = v[3][1] = v[6][1] = v[7][1] = y2;
205  v[0][2] = v[3][2] = v[4][2] = v[7][2] = z1;
206  v[1][2] = v[2][2] = v[5][2] = v[6][2] = z2;
207 
208  int drawtype;
209  switch (style) {
210  case STYLE_LINES: drawtype = GL_LINE_LOOP; break;
211  case STYLE_POINTS: drawtype = GL_POINTS; break;
212  default: drawtype = GL_QUADS; break;
213  }
214  for (int i=0; i<6; i++) {
215  if (textures[i] > 0) {
216  glEnable(GL_TEXTURE_2D);
217  glBindTexture(GL_TEXTURE_2D, textures[i]);
218  if ((rtx[i][0] > 1) || (rtx[i][1] > 1)) {
219  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
220  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
221  }
222  }
223  glBegin(drawtype);
224  for (int j=0; j<4; j++) {
225  if (textures[i] > 0)
226  glTexCoord2fv(&texcoords[i][j][0]);
227  glNormal3fv(&normals[i][0]);
228  vertex3fv(&v[faces[i][j]][0]);
229  }
230  glEnd();
231  if (textures[i] > 0)
232  glDisable(GL_TEXTURE_2D);
233  }
234 }

341 /*
342  * Torus drawing (taken from Mesa)
343  */
344 void Draw::torus(GLfloat rc, uint8_t numc, GLfloat rt, uint8_t numt, uint8_t style)
345 {
346  for (int i=0; i < numc; i++) {
347  float s, t, x, y, z;
348 
349  if (style == STYLE_FILL) glBegin(GL_QUAD_STRIP);
350  else glBegin(GL_LINE_STRIP);
351  for (int j=0; j <= numt; j++) {
352  for (int k=1; k >= 0; k--) {
353  s = (i + k) % numc + 0.5;
354  t = j % numt;
355  x = cos(t*M_2PI/numt) * cos(s*M_2PI/numc);
356  y = sin(t*M_2PI/numt) * cos(s*M_2PI/numc);
357  z = sin(s*M_2PI/numc);
358  glNormal3f(x, y, z);
359 
360  x = (rt + rc * cos(s*M_2PI/numc)) * cos(t*M_2PI/numt);
361  y = (rt + rc * cos(s*M_2PI/numc)) * sin(t*M_2PI/numt);
362  z = rc * sin(s*M_2PI/numc);
363  vertex3f(x, y, z);
364  }
365  }
366  glEnd();
367  }
368 }

370 /*
371  * Sphere (taken from Mesa)
372  */
373 void Draw::sphere(GLfloat r, uint8_t slices, uint8_t stacks, uint8_t style)
374 {
375  GLfloat rho, drho, theta, dtheta;
376  GLfloat x, y, z;
377  GLfloat s, t, ds, dt;
378 
379  drho = M_PI / (GLfloat) stacks;
380  dtheta = M_2PI / (GLfloat) slices;
381 
382  // draw +Z end as a triangle fan
383  glBegin(GL_TRIANGLE_FAN);
384  glNormal3f(0, 0, 1);
385  glTexCoord2f(0.5, 0);
386  vertex3f(0, 0, r);
387  for (int j=0; j <= slices; j++) {
388  theta = (j==slices) ? 0 : j*dtheta;
389  x = -sin(theta) * sin(drho);
390  y = cos(theta) * sin(drho);
391  z = cos(drho);
392  glNormal3f(x, y, z);
393  vertex3f(x*r, y*r, z*r);
394  }
395  glEnd();
396 
397  ds = 1. / slices;
398  dt = 1. / stacks;
399  t = 1; // because loop now runs from 0
400 
401  // draw intermediate stacks as quad strips
402  for (int i=0; i < stacks; i++) {
403  rho = i * drho;
404  glBegin(GL_QUAD_STRIP);
405  s = 0;
406  for (int j=0; j <= slices; j++) {
407  theta = (j==slices) ? 0 : j*dtheta;
408  x = -sin(theta) * sin(rho);
409  y = cos(theta) * sin(rho);
410  z = cos(rho);
411  glNormal3f(x, y, z);
412  glTexCoord2f(s, 1-t);
413  vertex3f(x*r, y*r, z*r);
414  x = -sin(theta) * sin(rho+drho);
415  y = cos(theta) * sin(rho+drho);
416  z = cos(rho+drho);
417  glNormal3f(x, y, z);
418  glTexCoord2f(s, 1-(t-dt));
419  s += ds;
420  vertex3f(x*r, y*r, z*r);
421  }
422  glEnd();
423  t -= dt;
424  }
425  // draw -Z end as a triangle fan
426  glBegin(GL_TRIANGLE_FAN);
427  glNormal3f(0, 0, -1);
428  glTexCoord2f(0.5, 1);
429  vertex3f(0, 0, -r);
430  rho = M_PI - drho;
431  s = 1;
432  t = dt;
433  for (int j=slices; j >= 0; j--) {
434  theta = (j==slices) ? 0 : j*dtheta;
435  x = -sin(theta) * sin(rho);
436  y = cos(theta) * sin(rho);
437  z = cos(rho);
438  glNormal3f(x, y, z);
439  glTexCoord2f(s, 1-t);
440  s -= ds;
441  vertex3f(x*r, y*r, z*r);
442  }
443  glEnd();
444 }

446 /*
447  * Cylinder (taken from Mesa, Brian Paul)
448  */
449 void Draw::cylinder(GLfloat baseR, GLfloat topR, GLfloat height, uint8_t slices, uint8_t stacks, uint8_t style)
450 {
451  GLfloat da, r, dr, dz;
452  GLfloat z, nz;
453  GLfloat ds = 1. / slices;
454  GLfloat dt = 1. / stacks;
455  GLfloat t = 0;
456 
457  da = M_2PI / slices;
458  dr = (topR-baseR) / stacks;
459  dz = height / stacks;
460  nz = (baseR-topR) / height; // Z component of normal vectors
461 
462  switch (style) {
463  case STYLE_FILL:
464  {
465  z = 0;
466  r = baseR;
467  for (int j=0; j < stacks; j++) {
468  GLfloat s = 0;
469  glBegin(GL_QUAD_STRIP);
470  for (int i=0; i <= slices; i++) {
471  GLfloat x, y;
472  if (i == slices) {
473  x = sin(0.0);
474  y = cos(0.0);
475  }
476  else {
477  x = sin(i*da);
478  y = cos(i*da);
479  }
480  glNormal3f(x, y, nz);
481  glTexCoord2f(s, t);
482  vertex3f(x*r, y*r, z);
483  glNormal3f(x, y, nz);
484  glTexCoord2f(s, t+dt);
485  vertex3f(x*(r+dr), y*(r+dr), z+dz);
486  s += ds;
487  }
488  glEnd();
489  r += dr;
490  t += dt;
491  z += dz;
492  }
493  }
494  break;
495  case STYLE_LINES:
496  {
497  z = 0;
498  r = baseR;
499  for (int j=0; j <= stacks; j++) {
500  glBegin(GL_LINE_LOOP);
501  for (int i=0; i < slices; i++) {
502  GLfloat x = cos(i*da);
503  GLfloat y = sin(i*da);
504  glNormal3f(x, y, nz);
505  glVertex3f(x*r, y*r, z);
506  }
507  glEnd();
508  z += dz;
509  r += dr;
510  }
511  /* draw length lines */
512  glBegin(GL_LINES);
513  for (int i=0;i<slices;i++) {
514  GLfloat x = cos(i*da);
515  GLfloat y = sin(i*da);
516  glNormal3f(x, y, nz);
517  glVertex3f(x*baseR, y*baseR, 0);
518  glVertex3f(x*topR, y*topR, height);
519  }
520  glEnd();
521  }
522  break;
523  case STYLE_POINTS:
524  {
525  glBegin(GL_POINTS);
526  for (int i=0; i < slices; i++) {
527  GLfloat x = cos(i*da);
528  GLfloat y = sin(i*da);
529  glNormal3f(x, y, nz);
530  z = 0;
531  r = baseR;
532  for (int j=0; j <= stacks; j++) {
533  glVertex3f(x*r, y*r, z);
534  z += dz;
535  r += dr;
536  }
537  }
538  glEnd();
539  }
540  break;
541  }
542 }

544 /*
545  * Disk (taken from Mesa, Brian Paul)
546  */
547 void Draw::disk(GLfloat innerR, GLfloat outerR, uint8_t slices, uint8_t loops, uint8_t style)
548 {
549  /* Normal vectors */
550  glNormal3f(0, 0, -1);
551 
552  GLfloat da = M_2PI / slices;
553  GLfloat dr = (outerR-innerR) / (GLfloat) loops;
554 
555  switch (style) {
556  case STYLE_FILL:
557  {
558  /* texture of a gluDisk is a cut out of the texture unit square
559  * x, y in [-outerR, +outerR]; s, t in [0, 1] * (linear mapping)
560  */
561  GLfloat dtc = 2. * outerR;
562  GLfloat sa, ca;
563  GLfloat r1 = innerR;
564  for (int l=0; l<loops; l++) {
565  GLfloat r2 = r1 + dr;
566  glBegin(GL_QUAD_STRIP);
567  for (int s=slices; s>=0 ; s--) {
568  GLfloat a;
569  if (s == slices)
570  a = 0;
571  else
572  a = s * da;
573  sa = sin(a);
574  ca = cos(a);
575  glTexCoord2f(0.5-sa*r2/dtc, 0.5+ca*r2/dtc);
576  glVertex3f(r2*sa, r2*ca, 0);
577  glTexCoord2f(0.5-sa*r1/dtc, 0.5+ca*r1/dtc);
578  glVertex3f(r1*sa, r1*ca, 0);
579  }
580  glEnd();
581  r1 = r2;
582  }
583  }
584  break;
585  case STYLE_LINES:
586  {
587  // draw loops
588  for (int l=0; l<=loops; l++) {
589  GLfloat r = innerR + l * dr;
590  glBegin(GL_LINE_LOOP);
591  for (int s=0; s<slices; s++) {
592  GLfloat a = s * da;
593  glVertex3f(r*sin(a), r*cos(a), 0);
594  }
595  glEnd();
596  }
597  // draw spokes
598  for (int s=0; s<slices; s++) {
599  GLfloat a = s * da;
600  GLfloat x = sin(a);
601  GLfloat y = cos(a);
602  glBegin(GL_LINE_STRIP);
603  for (int l=0; l<=loops; l++) {
604  GLfloat r = innerR + l * dr;
605  glVertex3f(r*x, r*y, 0);
606  }
607  glEnd();
608  }
609  }
610  break;
611  case STYLE_POINTS:
612  {
613  glBegin(GL_POINTS);
614  for (int s=0; s<slices; s++) {
615  GLfloat a = s * da;
616  GLfloat x = sin(a);
617  GLfloat y = cos(a);
618  for (int l=0; l<=loops; l++) {
619  GLfloat r = innerR * l * dr;
