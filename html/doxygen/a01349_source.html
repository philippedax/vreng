236  if (!strcasecmp(listObj[l]->typeName(), actiontype))

240  else {
241  if (nb == 1 && listObj[0]) {
242  notice("object found, do \"%s\" action !", val);
243  listObj[0]->runAction(val);

245  else if (nb > 1) {
246  notice("ambiguity, %d objects found, be more precise", nb);
247  }
248  else {

250  Vicinity *vicin = new Vicinity();
