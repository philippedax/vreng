44 void Wind::parser(char *l)
45 {
46  defaults();
47  l = tokenize(l);
48  begin_while_parse(l) {
49  l = parse()->parseAttributes(l, this);
50  if (!l) break;
51  if (!stringcmp(l, "speed")) l = parse()->parseUInt16(l, &speed, "speed");
52  else if (!stringcmp(l, "orient")) l = parse()->parseFloat(l, &orient, "orient");
53  }
54  end_while_parse(l);
55 }
