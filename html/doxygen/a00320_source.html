42 /* Downloads urls: filling of the two arrays */
43 void Book::httpReader(void *_book, Http *http)
44 {
45  Book *book = (Book *) _book;
46  if (! book) return;
47 
48  FILE *f = Cache::openCache(book->getUrl(), http);
49  if (! f) { error("can't open %s", book->getUrl()); return; }
50 
51  char line[BUFSIZ];
52 
53  // get the number of pages (first line)
54  if (! fgets(line, sizeof(line), f)) { error("can't get number of pages"); return; }
55  line[strlen(line) - 1] = '\0';
56  book->nbs = atoi(line) / 2;
57  if (book->nbs <= 0) { error("number of pages is negative"); return; }
58 
59  book->tex = new char*[2*book->nbs+3]; // +3 for edge
60  book->html = new char*[2*book->nbs];
61  if (! book->tex || ! book->html) { error("httpReader: can' alloc tex | html"); return; }
62 
63  int i;
64  char *p = NULL;
65 
66  for (i=0 ; i < 2*book->nbs ; i++) {
67  if (! fgets(line, sizeof(line), f)) return; // eof
68  line[strlen(line) - 1] = '\0';
69  if ((p = strtok(line, " \t")) == NULL) {
70  error("httpReader: bad line %s", line);
71  return;
72  }
73  // copy url of the html page matching this texture
74  char* sheet_tex = strdup(p);
75  book->tex[i] = strdup(p);
76  if ((p = strtok(NULL, " \t"))) {
77  char* sheet_html = strdup(p);
78  book->html[i] = strdup(p);
79  free(sheet_html);
80  }
81  else {
82  char* sheet_html = strdup(sheet_tex);
83  book->html[i] = strdup(sheet_tex);
84  free(sheet_html);
85  }
86  free(sheet_tex);
87  }
88 
89  // it remains 3 urls: textures for edge
90  if (! fgets(line, sizeof(line), f)) strcpy(line, DEF_URL_PAPER);
91  line[strlen(line) - 1] = '\0';
92  if ((p = strtok(line, " \t")) == NULL) strcpy(p, DEF_URL_PAPER);
93 
94  book->tex[i] = strdup(p);
95  char *tranche = strdup(p);
96  i++;
97  if ((p = strtok(NULL, " \t"))) book->tex[i] = strdup(p);
98  else book->tex[i] = strdup(tranche);
99  i++;
100  if ((p = strtok(NULL, " \t"))) book->tex[i] = strdup(p);
101  else book->tex[i] = strdup(tranche);
102  free(tranche);
103  File::closeFile(f);
104 }

168 void Book::parser(char *l)
169 {
170  defaults();
171  l = tokenize(l);
172  begin_while_parse(l) {
173  l = parse()->parseAttributes(l, this);
174  if (!l) break;
175  if (!stringcmp(l, "url=")) l = parse()->parseUrl(l, url);
176  else if (!stringcmp(l, "aright")) l = parse()->parseFloat(l, &aright, "aright");
177  else if (!stringcmp(l, "aleft")) l = parse()->parseFloat(l, &aleft, "aleft");
178  else if (!stringcmp(l, "nbsheets")) l = parse()->parseUInt8(l, &nbs, "nbsheets");
179  else if (!stringcmp(l, "sheetthickness")) l = parse()->parseFloat(l, &thick, "sheetthickness");
180  else if (!stringcmp(l, "currentsheet")) l = parse()->parseUInt8(l, &num, "currentsheet");
181  }
182  end_while_parse(l);
183 }

185 Book::Book(char *l)
186 {
187  parser(l);
188 
189  getDim(size);
190  width = size.v[0];
191  height = size.v[2];
192 
193  if (! *url) return;
194  if (Http::httpOpen(url, httpReader, this, 0) < 0) return;
195  if (! nbs) return; // no sheets
196 
197  enableBehavior(PERSISTENT);
198  enableBehavior(COLLIDE_NEVER);
199  enableBehavior(TAKABLE);
200  setRenderPrior(RENDER_HIGH);
201 
202  initializeMobileObject(1);
203  createPermanentNetObject(PROPS, ++oid);
204 
205  // create the heaps of sheets
206 
207  char s[BUFSIZ];
208  if (num == 0) { // book closed on the right side
209  state = CLOSED_R;
210  // create the right heap
211  float depth = thick * nbs; // epaisseur du tas droit
212  setPos(s, pos.x-depth/2*sin(aright), pos.y+depth/2*cos(aright), pos.z, aright, pos.ax);
213  setDim(s, width, depth, height);
214  setCov(s, tex[0], tex[2*nbs-1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
215  createSheet(s, Sheet::RIGHT, RIGHT);
216  // create the left heap (thick: 1 sheet)
217  setPos(s, pos.x, pos.y, pos.z, aright, pos.ax);
218  setDim(s, width, thick, height);
219  setCov(s, tex[0], tex[1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
220  createSheet(s, Sheet::RIGHT, LEFT);
221  }
222  else { // book closes on the left side
223  if (num == nbs) {
224  state = CLOSED_L;
225  // create the right heap (thick: 1 sheet)
226  setPos(s, pos.x, pos.y, pos.z, aleft, pos.ax);
227  setDim(s, width, thick, height);
228  setCov(s, tex[2*nbs-2],tex[2*nbs-1],tex[2*nbs],tex[2*nbs+2],tex[2*nbs+1]);
229  createSheet(s, Sheet::LEFT, RIGHT);
230  // create the left heap
231  float depth = thick * nbs; // epaisseur du tas gauche
232  setPos(s, pos.x+depth/2*sin(aleft), pos.y-depth/2*cos(aleft), pos.z, aleft, pos.ax);
233  setDim(s, width, depth, height);
234  setCov(s, tex[0], tex[2*nbs-2], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
235  createSheet(s, Sheet::LEFT, LEFT);
236  }
237  else { // book opened
238  state = OPENED;
239  // create the right heap
240  float depth = thick * (nbs - num); // right heap thick
241  float depth2 = depth + thick; // augmented
242  setPos(s, pos.x-depth2*sin(aright), pos.y+depth2*cos(aright), pos.z, aright, pos.ax);
243  setDim(s, width, depth, height);
244  setCov(s, tex[2*num],tex[2*nbs-1],tex[2*nbs],tex[2*nbs+2],tex[2*nbs+1]);
245  createSheet(s, Sheet::RIGHT, RIGHT);
246  // create the left heap
247  depth = thick * num; // demi-epaisseur du tas gauche
248  setPos(s,pos.x+(depth/2)*sin(aleft), pos.y-(depth/2)*cos(aleft), pos.z, aleft, pos.ax);
249  setDim(s, width, depth, height);
250  setCov(s, tex[0], tex[2*num-1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
251  createSheet(s, Sheet::LEFT, LEFT);
252  // create the inter sheet
253  setPos(s, pos.x, pos.y, pos.z, aright, pos.ax);
254  setDim(s, width, thick, height);
255  setCov(s, tex[2*num], tex[2*num+1]);
256  createSheet(s, Sheet::RIGHT, VOLATILE);
257  }
258  }

295 /* next page */
296 void Book::nextSheet(Book *book, void *d, time_t sec, time_t usec)
297 {
298  if (state == CLOSED_L) return;
299  char s[BUFSIZ];
300  if (num == nbs-1) { // cas 1: it remains only one sheet on the right
301  // increase the left heap minor edge size
302  float depth = thick * nbs; // new left heap thick
303  float depth2 = depth + 2.5*thick; // augmented
304  setPos(s, pos.x + depth2*sin(aleft), pos.y - depth2*cos(aleft), pos.z, aleft, pos.ax);
305  setDim(s, width, depth, height);
306  setTex(s, tex[0], tex[2*nbs-3], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
307  cancelSheet(left);
308  createSheet(s, Sheet::LEFT, LEFT);
309  // destruction de la inter sheet
310  cancelSheet(inter);
311  inter = NULL;
312  // rotation de la partie droite
313  turnNext(right, d, sec, usec);
314  state = CLOSED_L;
315  }
316  else {
317  if (num == 0) { // cas 2: none sheet one the left, inter is NULL
318  // on cree la inter mais on fait tourner la partie gauche (resumee a 1 feuille)
319  // create inter sheet first
320  setPos(s, pos.x, pos.y, pos.z, aright, pos.ax);
321  setDim(s, width, thick, height);
322  setTex(s, tex[2], tex[3]);
323  createSheet(s, Sheet::LEFT, VOLATILE);
324  // replace the right heap
325  float depth = thick * (nbs-1); // right heap thick
326  float depth2 = depth + thick; // augmented
327  setPos(s, pos.x - depth2*sin(aright), pos.y + depth2*cos(aright), pos.z, aright, pos.ax);
328  setDim(s, width, depth, height);
329  setTex(s, tex[2], tex[2*nbs-1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]); // p 2 + couverture arrire
330  cancelSheet(right);
331  createSheet(s, Sheet::RIGHT, RIGHT);
332  // replace the left heap if too big (case of bookClose)
333  if (left) left->getDim(size);
334  else { error("no left sheet"); return; }
335  if (size.v[1] > 0.7 * thick) {
336  setPos(s, pos.x, pos.y, pos.z, aright, pos.ax);
337  setDim(s, width, thick, height);
338  setTex(s, tex[0], tex[1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]); // couverture avant + p 1
339  cancelSheet(left);
340  createSheet(s, Sheet::RIGHT, LEFT);
341  }
342  // turn the heap (1 sheet) left
343  turnNext(left, d, sec, usec);
344  state = OPENED;
345  }
346  else { // general case
347  // replace the left heap
348  float depth = thick * (num+1); // new left heap thick
349  float depth2 = depth + thick; // augmented
350  setPos(s, pos.x + depth2*sin(aleft), pos.y - depth2*cos(aleft), pos.z, aleft, pos.ax);
351  setDim(s, width, depth, height);
352  setTex(s, tex[0], tex[2*num-1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
353  cancelSheet(left);
354  createSheet(s, Sheet::LEFT, LEFT);
355  // replacement of inter sheet if needed
356  if (! inter) { error("no inter sheet"); return; }
357  if (inter->state == Sheet::LEFT) {
358  setPos(s, pos.x, pos.y, pos.z, aright, pos.ax);
359  setDim(s, width, thick, height);
360  setTex(s, tex[2*num], tex[2*num+1]);
361  cancelSheet(inter);
362  createSheet(s, Sheet::RIGHT, VOLATILE);
363  }
364  // replace the right heap
365  depth = thick * (nbs-(num+1));//right heap thick
366  depth2 = depth + thick; // augmented
367  setPos(s, pos.x - depth2*sin(aright), pos.y + depth2*cos(aright), pos.z, aright, pos.ax);
368  setDim(s, width, depth, height);
369  setTex(s, tex[2*(num+1)], tex[2*nbs-1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
370  cancelSheet(right);
371  createSheet(s, Sheet::RIGHT, RIGHT);
372  // rotation of inter sheet
373  turnNext(inter, d, sec, usec);
374  }
375  }

379 void Book::prevSheet(Book *book, void *d, time_t sec, time_t usec)
380 {
381  if (state == CLOSED_R) return;
382  char s[BUFSIZ];
383  if (num == 1) { // cas 1: it remains only one sheet on the left
384  // increase the right heap minor edge size
385  float depth = thick * nbs; // right heap thick
386  float depth2 = depth + 2.5*thick; // augmented
387  setPos(s, pos.x - depth2*sin(aright), pos.y + depth2*cos(aright), pos.z, aright, pos.ax);
388  setDim(s, width, depth, height);
389  setTex(s, tex[2], tex[2*nbs-1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
390  cancelSheet(right);
391  createSheet(s, Sheet::RIGHT, RIGHT);
392  // destroy inter sheet
393  cancelSheet(inter);
394  inter = NULL;
395  // rotation left sheet
396  turnPrev(left, d, sec, usec);
397  state = CLOSED_R;
398  }
399  else {
400  if (num == nbs) { // cas 2: none sheet on the right
401  // cree la inter mais on fait tourner la partie droite: 1 feuille
402  // create the inter sheet
403  setPos(s, pos.x, pos.y, pos.z, aleft, pos.ax);
404  setDim(s, width, thick, height);
405  setTex(s, tex[2*nbs-4], tex[2*nbs-3]); // p 2N-4 + p 2N-3
406  createSheet(s, Sheet::RIGHT, VOLATILE);
407  // replace the left heap
408  float depth = thick * (nbs-1); // new left heap thick
409  float depth2 = depth + thick; // augmented
410  setPos(s, pos.x + depth2*sin(aleft), pos.y - depth2*cos(aleft), pos.z, aleft, pos.ax);
411  setDim(s, width, depth, height);
412  setTex(s, tex[0], tex[2*nbs-3], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]); // p 0 + p 2N-3
413  cancelSheet(left);
414  createSheet(s, Sheet::LEFT, LEFT);
415  // replace the right heap if not too big
416  if (right) right->getDim(size);
417  else { error("no right sheet"); return; }
418  if (size.v[1] > 0.7 * thick) {
419  setPos(s, pos.x, pos.y, pos.z, aright, pos.ax);
420  setDim(s, width, thick, height);
421  setTex(s, tex[2*nbs-2], tex[2*nbs-1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
422  cancelSheet(right);
423  createSheet(s, Sheet::LEFT, RIGHT);
424  }
425  // turn the heap of 1 sheet on the right
426  turnPrev(right, d, sec, usec);
427  state = OPENED;
428  }
429  else { // general case
430  // replace the right heap
431  float depth = thick * (nbs-num+1); // right heap thick
432  float depth2 = depth + thick; // augmented
433  setPos(s, pos.x - depth2*sin(aright), pos.y + depth2*cos(aright), pos.z, aright, pos.ax);
434  setDim(s, width, depth, height);
435  setTex(s, tex[2*num], tex[2*nbs-1], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
436  cancelSheet(right);
437  createSheet(s, Sheet::RIGHT, RIGHT);
438  // replacement of inter sheet if needed
439  if (! inter) { error("no inter sheet"); return; }
440  if (inter->state == Sheet::RIGHT) {
441  setPos(s, pos.x, pos.y, pos.z, aleft, pos.ax);
442  setDim(s, width, thick, height);
443  setTex(s, tex[2*(num-1)],tex[2*num-1]);
444  cancelSheet(inter);
445  createSheet(s, Sheet::LEFT, VOLATILE);
446  }
447  // replace the left heap
448  depth = thick * (num -1); // new left heap thick
449  depth2 = depth + thick; // augmented
450  setPos(s, pos.x + depth2*sin(aleft), pos.y - depth2*cos(aleft), pos.z, aleft, pos.ax);
451  setDim(s, width, depth, height);
452  setTex(s, tex[0], tex[2*num-3], tex[2*nbs], tex[2*nbs+2], tex[2*nbs+1]);
453  cancelSheet(left);
454  createSheet(s, Sheet::LEFT, LEFT);
455  // rotate inter sheet
456  turnPrev(inter, d, sec, usec);
457  }
458  }

583 /* approach the book near the avatar */
584 void Book::approach()
585 {
586  float dx = .7 * (localuser->pos.x - pos.x); // not too near
587  float dy = .7 * (localuser->pos.y - pos.y); // not too near
588  float dz = localuser->pos.z + localuser->height/2 -.10 - pos.z;
589  float aright0 = aright;
590  float aleft0 = aleft;
591  float azend = localuser->pos.az - M_PI_2; // note: azend = "aright_fin"
592  float axend = localuser->pos.ax;
593  float ayend = localuser->pos.ay;
594  float daz = deltaAngle(azend, aright0);
595  float dax = deltaAngle(axend, pos.ax);
596  float day = deltaAngle(ayend, pos.ay);
597 
598  float ttl = ABSF(dx / (LSPEED * 2));
599  if (ttl == 0) ttl = ABSF(dy / (LSPEED * 2));
600 
601  if (ttl > 0.001) {
602  float vz = dz / ttl;
603  float vaz = daz / ttl;
604  float vax = dax / ttl;
605  float vay = day / ttl;
606 
607  // move the book
608  move.lspeed.v[0] = dx / ttl;
609  move.lspeed.v[1] = dy / ttl;
610  move.lspeed.v[2] = vz;
611  move.aspeed.v[0] = vaz;
612  move.aspeed.v[1] = vay;
613  move.aspeed.v[2] = vax;
614  aright = azend;
615  aleft += daz;
616 
617  // move the right sheet
618  if (right) {
619  right->getDim(size);
620  float dxright = dx + size.v[1] * (sin(aright0) - sin(azend));
621  float dyright = dy + size.v[1] * (cos(azend) - cos(aright0));
622  right->move.lspeed.v[0] = dxright / ttl;
623  right->move.lspeed.v[1] = dyright / ttl;
624  right->move.lspeed.v[2] = vz;
625  right->move.aspeed.v[0] = vaz;
626  right->move.aspeed.v[1] = vay;
627  right->move.aspeed.v[2] = vax;
628  right->aright = azend;
629  right->aleft += daz;
630  }
631  // move the left sheet
632  if (left) {
633  left->getDim(size);
634  float dxleft = dx + size.v[1] * (sin(aleft0+daz) - sin(aleft0));
635  float dyleft = dy + size.v[1] * (cos(aleft0) - cos(aleft0+daz));
636  left->move.lspeed.v[0] = dxleft / ttl;
637  left->move.lspeed.v[1] = dyleft / ttl;
638  left->move.lspeed.v[2] = vz;
639  left->move.aspeed.v[0] = vaz;
640  left->move.aspeed.v[1] = vay;
641  left->move.aspeed.v[2] = vax;
642  left->aright = azend;
643  left->aleft += daz;
644  }
645  // move the inter sheet
646  if (inter) {
647  inter->move.lspeed.v[0] = dx / ttl;
648  inter->move.lspeed.v[1] = dy / ttl;
649  inter->move.lspeed.v[2] = vz;
650  inter->move.aspeed.v[0] = vaz;
651  inter->move.aspeed.v[1] = vay;
652  inter->move.aspeed.v[2] = vax;
653  inter->aright = azend;
654  inter->aleft += daz;
655  inter->initImposedMovement(ttl);
656  }
657  if (left) left->initImposedMovement(ttl);
658  if (right) right->initImposedMovement(ttl);
659  initImposedMovement(ttl);
660  }
661  near = true;
662 }
